<script>
const krwInput = document.getElementById('krw');
const rateInput = document.getElementById('rate');
const taxSelect = document.getElementById('tax');
const weightInput = document.getElementById('weight');

function calculate() {
  const krw = Number(krwInput.value) || 0;
  const rate = Number(rateInput.value) || 46;
  const tax = Number(taxSelect.value);
  const weight = Number(weightInput.value) || 0;

  const twd = krw / rate;
  const afterTax = krw * (1 - tax);
  const shipping = weight * 5.5;
  const total = afterTax + shipping;
  const totalTwd = total / rate;

  document.getElementById('twd').innerText = `≈ NT$ ${twd.toFixed(0)}`;
  document.getElementById('afterTax').innerText = `退稅後：${afterTax.toFixed(0)} 韓元`;
  document.getElementById('shipping').innerText = `運費：${shipping.toFixed(0)} 韓元`;
  document.getElementById('totalKrw').innerText = `${total.toFixed(0)} 韓元`;
  document.getElementById('totalTwd').innerText = `≈ NT$ ${totalTwd.toFixed(0)}`;
}

function resetAll() {
  krwInput.value = '';
  weightInput.value = '';
  calculate();
}

document.querySelectorAll('input, select').forEach(el => {
  el.addEventListener('input', calculate);
});
</script>
